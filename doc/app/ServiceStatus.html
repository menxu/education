def check_faye
  if !faye_is_run?
    raise 'faye 没有启动，可以使用 ./deploy/sh/faye.sh start 命令启动 faye'
  end
end

def faye_is_run?
  sh_path = File.join(ROOT_PATH, 'deploy/sh/faye.sh')
  _service_is_run?(sh_path)
end
